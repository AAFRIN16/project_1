    <script>
        function addTask() {
            let input = document.getElementById("taskInput");
            if (input.value.trim() === "") return;

            let li = document.createElement("li");
            li.textContent = input.value;

            let removeBtn = document.createElement("button");
            removeBtn.textContent = "X";
            removeBtn.className = "remove";
            removeBtn.onclick = function () {
                li.remove();
            };

            li.appendChild(removeBtn);
            document.getElementById("taskList").appendChild(li);
            input.value = "";
        }
    </script>

    let text = document.getElementById("noteInput").value.trim();

        }
        #result { margin-top: 10px; font-weight: bold; }
    </style>
</head>
<body>

<h2>Temperature Converter</h2>

<input type="number" id="temp" placeholder="Enter value">
<select id="unit">
    <option value="CtoF">Celsius ‚Üí Fahrenheit</option>
    <option value="FtoC">Fahrenheit ‚Üí Celsius</option>
</select>

<br>
<button onclick="convert()">Convert</button>

<p id="result"></p>

<script>
function convert() {
    let temp = parseFloat(document.getElementById("temp").value);
    let unit = document.getElementById("unit").value;
    let result = 0;

    if (unit === "CtoF") {
        result = (temp * 9/5) + 32;
        document.getElementById("result").textContent = result.toFixed(2) + " ¬∞F";
    } else {
        result = (temp - 32) * 5/9;
        document.getElementById("result").textContent = result.toFixed(2) + " ¬∞C";
    }
}
</script>

</body>
</html>
